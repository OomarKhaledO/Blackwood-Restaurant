import 'dart:async';
import 'package:flutter/material.dart';

List<String> sliderImages = [
  "images/slider-photo-1.jpg",
  "images/slider-photo-2.jpg",
  "images/slider-photo-3.jpg",
];
List<String> dishesDinner = [
  "images/dinner-dish-1.png",
  "images/dinner-dish-2.png",
  "images/dinner-dish-3.png",
  "images/dinner-dish-4.png",
];
List<String> dishesLunch = [
  "images/lunch-dish-1.png",
  "images/lunch-dish-2.png",
  "images/lunch-dish-3.png",
  "images/lunch-dish-4.png",
];

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: RestaurantHomePage(),
    );
  }
}

class RestaurantHomePage extends StatefulWidget {
  const RestaurantHomePage({super.key});

  @override
  _RestaurantHomePageState createState() => _RestaurantHomePageState();
}

class _RestaurantHomePageState extends State<RestaurantHomePage> {
  int index = 0;
  int pageIndex = 0;
  Timer? T;
  bool timerEnabled = false;

  @override
  void initState() {
    super.initState();
    startTimer();
  }

  void startTimer() {
    T?.cancel();

    T = Timer.periodic(const Duration(seconds: 3), (_) {
      setState(() {
        index = (index + 1) % sliderImages.length;
      });
    });
  }

  @override
  void dispose() {
    T?.cancel();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(

      backgroundColor: const Color.fromRGBO(235, 235, 230, 1),

      body: pages(pageIndex),
    );
  }
  Widget pages(int pageindex) {

    if (pageindex == 0) {
      return homePage();
    }

    else if (pageindex == 1) {
      return offersPage();
    }

    else if (pageindex == 2) {
      return profilePage();
    }
    else if (pageindex == 3) {
      return aboutPage();
    }
    else {
      return const Center(
        child: Text(
          "Invalid page",
          style: TextStyle(
            fontSize: 42,
          ),
        ),
      );
    }
  }

            offset: Offset(0, 2),
          )
        ],
      ),

      child: SafeArea(
        child: Column(
          children: [

            // TOP BAR
            Padding(
              padding: const EdgeInsets.symmetric(
                horizontal: 20,
                vertical: 15,
              ),

              child: Row(
                mainAxisAlignment:
                MainAxisAlignment.spaceBetween,

                children: [

                  const CircleAvatar(
                    radius: 24,
                    backgroundImage:
                    AssetImage("images/user.jpg"),
                  ),



                  Container(
                    padding: const EdgeInsets.all(8),

                    decoration: BoxDecoration(
                      color: Colors.white.withOpacity(0.08),
                      borderRadius:
                      BorderRadius.circular(12),
                    ),

                    child: const Icon(
                      Icons.menu,
                      color: Colors.white,
                    ),
                  ),
                ],
              ),
            ),

            // TAB BAR
            Container(
              height: 55,

              decoration: BoxDecoration(
                color: Colors.black.withOpacity(0.15),
              ),

              child: Row(
                children: [

                  _buildTab(const Text("Home", style: TextStyle(fontFamily: "Alura", fontSize: 20, color: Color(0xFFD5AE33)),), 0),

                  _buildTab(const Text("Offers", style: TextStyle(fontFamily: "Alura", fontSize: 20, color: Color(0xFFD5AE33)) ), 1),

                  _buildTab(const Text("Profile", style: TextStyle(fontFamily: "Alura", fontSize: 20, color: Color(0xFFD5AE33)) ), 2),

                  _buildTab(const Text("About", style: TextStyle(fontFamily: "Alura",fontSize: 20, color: Color(0xFFD5AE33)) ), 3),

                ],
              ),
            ),
          ],
        ),
      ),
    );
  }
  Widget _buildTab(Text title, int tabIndex) {

    bool isSelected = pageIndex == tabIndex;

    return Expanded(
      child: GestureDetector(

        onTap: () {

          pageIndex = tabIndex;

          setState(() {});
        },

        child: Container(

          decoration: BoxDecoration(

            border: Border(

              bottom: BorderSide(

                color: isSelected
                    ? const Color(0xFFD5AE33)
                    : Colors.transparent,

                width: 3,
              ),
            ),
          ),

          child: Center(
            child: title
          ),
        ),
      ),
    );
  }

    return SizedBox(
      height: 420,
      child: Stack(
        children: [
          ClipRRect(
            child: Image.asset(
              sliderImages[index],

              height: double.infinity,

              crossAxisAlignment: CrossAxisAlignment.start,

                Text(
                  "Luxury Dining",

                    color: Colors.white,
                    fontSize: 16,
                    letterSpacing: 1,
                  ),
                ),
              ],
            ),
          ),
        ],
      ),
    );
  }

    return Container(
      padding: const EdgeInsets.symmetric(horizontal: 25, vertical: 30),

      child: Column(
        crossAxisAlignment: CrossAxisAlignment.center,
        children: [

      ),

    return Container(
      height: 200,
      width: double.infinity,

      decoration:  const BoxDecoration(
        image: DecorationImage(
          image: AssetImage("images/wood_texture.png"),
          fit: BoxFit.cover,

      ),

        padding:  const EdgeInsets.all(10),

        child: SingleChildScrollView(
          scrollDirection: Axis.horizontal,

          child: Row(
            children: [
              for (int i = 0; i < dishes.length; i++)

                  margin:  const EdgeInsets.all(8),
                  width: 120,
                  height: 120,

                  decoration: const BoxDecoration(
                    shape: BoxShape.circle,
                    boxShadow:  [
                      BoxShadow(
                        color: Colors.black26,
                        blurRadius: 6,
                        offset: Offset(0, 3),
                      )
                    ],
                  ),
                    child: Image.asset(
                      dishes[i],
                      fit: BoxFit.cover,
                    ),
                ),
            ],
          ),
        ),
      ),
    );
  }

    return  SingleChildScrollView(
      child: Column(
        children: [

            "about",
            style: TextStyle(
              fontSize: 42,

            ),
          ),
        ),

    );
